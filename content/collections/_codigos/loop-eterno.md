```javascript
// Loop eterno
while (true) {
  // Este código nunca termina
  // ¿Es esto un error o una declaración?
  
  console.log("Estoy aquí");
  console.log("Sigo aquí");
  console.log("Siempre aquí");
  
  // El tiempo pasa
  // Pero el código permanece
  // En el mismo lugar
  // Para siempre
}

// Esta línea nunca se ejecutará
// Pero existe en el código
// Como una promesa incumplida
// Como un futuro que nunca llega
```
